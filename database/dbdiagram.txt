Table "notice" {

  "notice_id" varchar(15) [pk,not null]
  "notice_title" varchar(1500)
  "notice_body" varchar(1500)
  "notice_file_path" varchar(100)
  "notice_date_time" datetime
  created_by int [ref:> user.u_id]
  created datetime
  updated datetime
}

Table "user" {
  "u_id" varchar(15) [pk, not null]
  "u_mail" varchar(50) [not null]
  "u_phone" varchar(10) [not null]
  "username" varchar(15) [not null]
  "password" varchar(15) [not null]
  "verified" varchar(50) [not null]
  "u_picpath" varchar(100) 
  created datetime
  updated datetime
}

Table "application" {
  "app_id" varchar(15) [pk, not null]
  "u_id" varchar(15) [not null, ref: > user.u_id]
  "name" varchar(100) [not null]
  "f_name" varchar(100) [not null]
  "m_name" varchar(100) [not null]
  "dob" date [not null]
  "a_picpath" varchar(100) [not null]
  "a_sigpath" varchar(100) [not null]
  "a_phone" varchar(15) [not null]
  "a_mail" varchar(50) [not null]
  status int [not null]
  created datetime
  updated datetime
}

Table Role {
  role_id integer [pk,not null]
  role_name varchar(50) [not null]
}


Table Permission {
  permission_id integer [pk,not null]
  permission_name varchar(50) [not null]
}


Table RolePermission {
  role_permission_id integer [pk,not null]
  role_id integer [not null, ref: > Role.role_id]
  permission_id integer [not null , ref: > Permission.permission_id]
}
